void T440_FeatureExtractor::_computeBasicFeatures(const float* p_signal, uint32_t p_len, SmeaType::FeatureSlot& p_slot) {
    DynamicConfig v_cfg = T415_ConfigManager::getInstance().getConfig();

    // [패치] 하드코딩되었던 42, 420 샘플을 동적 ms 기반으로 계산
    uint32_t v_staSamples = (uint32_t)(v_cfg.feature.sta_ms * SmeaConfig::System::SAMPLING_RATE_CONST / 1000.0f);
    uint32_t v_ltaSamples = (uint32_t)(v_cfg.feature.lta_ms * SmeaConfig::System::SAMPLING_RATE_CONST / 1000.0f);

    float v_sumSq = 0.0f, v_peak = 0.0f, v_staEnergy = 0.0f, v_ltaEnergy = 0.0f;
    float v_sum = 0.0f, v_varianceSum = 0.0f, v_moment4Sum = 0.0f;

    for (uint32_t i = 0; i < p_len; i++) v_sum += p_signal[i];
    float v_mean = v_sum / p_len;

    for (uint32_t i = 0; i < p_len; i++) {
        float v_val = fabsf(p_signal[i]);
        v_sumSq += v_val * v_val;
        if (v_val > v_peak) v_peak = v_val;
        
        if (i < v_staSamples) v_staEnergy += v_val * v_val;
        if (i < v_ltaSamples) v_ltaEnergy += v_val * v_val;

        float v_diff = p_signal[i] - v_mean;
        v_varianceSum += v_diff * v_diff;
        v_moment4Sum += v_diff * v_diff * v_diff * v_diff;
    }

    p_slot.rms = sqrtf(fmaxf(v_sumSq / p_len, 1e-12f));
    p_slot.energy = v_sumSq / (float)SmeaConfig::System::SAMPLING_RATE_CONST;
    p_slot.crest_factor = p_slot.rms > 0 ? (v_peak / p_slot.rms) : 0.0f;
    
    // [패치] LTA/STA 시간 배율도 동적 계산
    float v_ratio = v_cfg.feature.lta_ms / fmaxf(v_cfg.feature.sta_ms, 0.1f);
    p_slot.sta_lta_ratio = v_ltaEnergy > 0 ? (v_staEnergy / (v_ltaEnergy / v_ratio)) : 1.0f;

    // ... (나머지 동일) ...
}

**1. T440_FeatExtra (특징점 추출부) 적발 내역**
 * **STA / LTA (단기/장기 에너지) 윈도우 크기:** if (i < 42), if (i < 420)
   * **문제:** 42kHz 샘플링 기준 1ms, 10ms를 하드코딩한 아주 위험한 매직넘버입니다. 현장 임펄스 노이즈 길이에 따라 반드시 튜닝되어야 합니다.
   * **개선:** STA_MS_DEF (1.0f), LTA_MS_DEF (10.0f) 로 동적 변수화.
 * **공간 위상(IPD) 주파수 대역:** 100.0f ~ 4000.0f
   * **문제:** 위상 결맞음 분석을 수행할 주파수 대역이 고정되어 있어 특정 설비의 베어링 마찰음 대역 등을 유연하게 포착할 수 없습니다.
   * **개선:** IPD_FREQ_MIN_DEF, IPD_FREQ_MAX_DEF 로 동적 변수화.
 * **Cepstrum 분석 타겟 주기:** float v_targets[4] = {0.00833f, 0.01666f, ...}
   * **문제:** 특정 모터의 회전수(RPM)에 의존적인 퀘프런시(Quefrency) 타겟이 하드코딩되어 있습니다. (120Hz, 60Hz 등)
   * **개선:** CEPS_TARGETS_DEF[4] 배열로 분리하여 웹에서 타겟 주기를 설정 가능하게 변경.
**2. T450_FsmMgr (판정 및 FSM 제어부) 적발 내역**
 * **STA/LTA 임펄스 오탐지 임계치:** if (v_slot.sta_lta_ratio > 3.0f)
   * **문제:** 3.0 이라는 판정 기준이 숨어있어 런타임 튜닝이 불가능합니다.
   * **개선:** RULE_STA_LTA_THRESH_DEF 동적 변수화.
 * **교차 검증 반복 횟수:** if (v_this->_currentTrialNo < 3)
   * **문제:** 3회 검증이라는 시나리오가 하드코딩되어 있어, 빠른 판정이 필요한 곳에서는 병목이 될 수 있습니다.
   * **개선:** MAX_TRIAL_COUNT_DEF 동적 변수화.
