

#### ⚠️ 아직 코드에 반영해야 할 스펙 축소/누락 사항
*이 항목들은 "컴파일이나 시스템 다운"을 유발하진 않지만, **기계 결함 판정력을 떨어뜨리고 명세서를 위반하는 도메인 로직들**입니다. 이 부분들의 패치를 진행해야 합니다.*
 1. **[스펙 기만] 42kHz 대역폭 절단 (FIR LPF 컷오프):**
   * **문제:** 42kHz로 샘플링해 놓고 DSP 엔진에서 3.8kHz 이상을 다 잘라버리고 있습니다 (FIR_LPF_CUTOFF = 3800.0f).
   * **조치 필요:** 컷오프를 20,000.0f (20kHz)로 개방해야 합니다.
 2. **[기능 누락] 120Hz 주요 전원 고조파(Harmonics) 노치 필터 누락:**
   * **문제:** 60Hz 전원 노이즈만 잡고, 더 치명적인 1차 고조파인 120Hz 필터링이 누락되었습니다.
   * **조치 필요:** T430_DspEng에 120Hz Biquad Notch 필터 직렬(Cascade) 추가.
 3. **[수학 축소] MFCC Delta N=2 정밀 공식 간소화:**
   * **문제:** 명세서는 전후 2프레임(N=2)의 가중합을 요구했으나, 코드는 단순 전후 1프레임 차분으로 축소 구현되었습니다.
   * **조치 필요:** T440의 Delta 및 Delta-Delta 계산식에 N=2 수식 적용.
 4. **[기능 누락] 주파수 영역 'Band RMS' 피처 누락:**
   * **문제:** 베어링 결함 등 특정 주파수 대역의 에너지를 보는 Band RMS 지표가 구조체와 추출기에서 증발했습니다.
   * **조치 필요:** FeatureSlot에 배열 추가 및 T440 추출 로직 구현.
 5. **[기능 축소] 3회 반복 수집(워밍업 -> 교차 검증) 시나리오 증발:**
   * **문제:** 트리거 발생 시 1회성으로 수집하고 끝나는 단발성 구조입니다. (1회차 배경 학습, 2/3회차 본 검사 누락).
   * **조치 필요:** FSM 상태 머신에 시퀀스 루프 카운터(trial_no) 도입 여부 결정.
 6. **[판정 누락] Level 1 (Test NG - 마이크/신호 단선) 방어 누락:**
   * **문제:** 에너지가 0에 수렴하는 하드웨어 고장 상황을 걸러내는 1차 검문소가 없습니다.
   * **조치 필요:** runHybridDecision 최상단에 하한선(Lower bound) 예외 처리 추가.

bool T430_DspEngine::init() {
    // ... (FIR 초기화 유지) ...

    // Notch 필터 1 (60Hz) 계수 계산
    float v_omega = 2.0f * (float)M_PI * SmeaConfig::Dsp::NOTCH_FREQ_HZ / SmeaConfig::SAMPLING_RATE;
    float v_alpha = sinf(v_omega / SmeaConfig::Dsp::NOTCH_Q_FACTOR) / 2.0f;
    float v_a0 = 1.0f + v_alpha;
    v_notchCoeffs[0] = 1.0f / v_a0;
    v_notchCoeffs[1] = (-2.0f * cosf(v_omega)) / v_a0;
    v_notchCoeffs[2] = 1.0f / v_a0;
    v_notchCoeffs[3] = (-2.0f * cosf(v_omega)) / v_a0;
    v_notchCoeffs[4] = (1.0f - v_alpha) / v_a0;

    // [패치 2] Notch 필터 2 (120Hz) 계수 계산 추가
    float v_omega2 = 2.0f * (float)M_PI * SmeaConfig::Dsp::NOTCH_FREQ_2_HZ / SmeaConfig::SAMPLING_RATE;
    float v_alpha2 = sinf(v_omega2 / SmeaConfig::Dsp::NOTCH_Q_FACTOR) / 2.0f;
    float v_a0_2 = 1.0f + v_alpha2;
    v_notch2Coeffs[0] = 1.0f / v_a0_2;
    v_notch2Coeffs[1] = (-2.0f * cosf(v_omega2)) / v_a0_2;
    v_notch2Coeffs[2] = 1.0f / v_a0_2;
    v_notch2Coeffs[3] = (-2.0f * cosf(v_omega2)) / v_a0_2;
    v_notch2Coeffs[4] = (1.0f - v_alpha2) / v_a0_2;

    resetFilterStates();
    return true;
}

// [패치 4] Band RMS 구현부 추가
void T440_FeatureExtractor::computeBandRMS(SmeaType::FeatureSlot& p_slot) {
    float v_binRes = (float)SmeaConfig::SAMPLING_RATE / SmeaConfig::FFT_SIZE;
    uint16_t v_maxBin = (SmeaConfig::FFT_SIZE / 2) + 1;

    for (int b = 0; b < SmeaConfig::Feature::BAND_RMS_COUNT; b++) {
        float v_startHz = SmeaConfig::Feature::BAND_RANGES[b][0];
        float v_endHz = SmeaConfig::Feature::BAND_RANGES[b][1];
        
        int v_startBin = (int)(v_startHz / v_binRes);
        int v_endBin = (int)(v_endHz / v_binRes);
        
        float v_sumSq = 0.0f;
        int v_count = 0;
        
        for (int i = v_startBin; i <= v_endBin && i < v_maxBin; i++) {
            v_sumSq += v_powerSpectrum[i]; 
            v_count++;
        }
        p_slot.band_rms[b] = v_count > 0 ? sqrtf(v_sumSq / v_count) : 0.0f;
    }
}

// [패치 3] 명세서 N=2 수식 적용 (수학적 축소 복원)
void T440_FeatureExtractor::applyTemporalDerivatives(SmeaType::FeatureSlot& p_slot) {
    float* p_mfcc39 = p_slot.mfcc;
    uint16_t v_dim = SmeaConfig::MFCC_COEFFS;
    
    // MFCC 히스토리 밀어내기
    if (v_historyCount < 5) {
        memcpy(v_mfccHistory[v_historyCount], p_mfcc39, sizeof(float) * v_dim);
        // ... (RMS 히스토리 생략/유지) ...
        v_historyCount++;
    } else {
        for (int i = 0; i < 4; i++) memcpy(v_mfccHistory[i], v_mfccHistory[i + 1], sizeof(float) * v_dim);
        memcpy(v_mfccHistory[4], p_mfcc39, sizeof(float) * v_dim);
        // ... (RMS 히스토리 생략/유지) ...
    }

    if (v_historyCount >= 5) {
        // [패치 3] N=2 가중합 공식: d[t] = ((c[t+1] - c[t-1]) + 2*(c[t+2] - c[t-2])) / 10.0f
        for (int i = 0; i < v_dim; i++) {
            float v_delta = ((v_mfccHistory[3][i] - v_mfccHistory[1][i]) + 2.0f * (v_mfccHistory[4][i] - v_mfccHistory[0][i])) / 10.0f;
            p_mfcc39[v_dim + i] = v_delta;
        }

        // Delta-Delta를 위한 Delta 히스토리 밀어내기
        for (int i = 0; i < 4; i++) memcpy(v_deltaHistory[i], v_deltaHistory[i + 1], sizeof(float) * v_dim);
        memcpy(v_deltaHistory[4], p_mfcc39 + v_dim, sizeof(float) * v_dim);

        // [패치 3] N=2 기반 Delta-Delta 연산
        for (int i = 0; i < v_dim; i++) {
            float v_deltaDelta = ((v_deltaHistory[3][i] - v_deltaHistory[1][i]) + 2.0f * (v_deltaHistory[4][i] - v_deltaHistory[0][i])) / 10.0f;
            p_mfcc39[v_dim * 2 + i] = v_deltaDelta;
        }
        // ... (RMS Delta 유지) ...
    } else {
        memset(p_mfcc39 + v_dim, 0, v_dim * 2 * sizeof(float));
    }
}
